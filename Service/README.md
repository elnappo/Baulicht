# Service
## Install
    $ echo "export DISPAY=:0" > /etc/profile
    $ sudo apt-get update
    $ sudo apt-get install libqt4-dev
    $ sudo usermod -aG gpio pi
    $ cd Baulicht/Service
    $ qmake -r
    $ make
    $ sudo make install
